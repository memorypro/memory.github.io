    void LCD_Send( uint8_t  type, uint8_t transdata )
{
    uint8_t firstbyte = 0xf8;
    uint8_t temp;
    uint8_t i, j = 3;
    if( type )
    {
        firstbyte |= 0x02;
    }
    LCD_CS_PORT |= LCD_CS_PIN;
    LCD_CLK_PORT &= ~LCD_CLK_PIN;
    while( j > 0 )
    {
        if( j == 3 )
        {
            temp = firstbyte;
        }
        else if( j == 2 )
        {
            temp = transdata & 0xf0;
        }
        else
        {
            temp = ( transdata << 4 ) & 0xf0;
        }
        for( i = 8; i > 0; i-- )
        {
            if( temp & 0x80 )
            {
                LCD_SDI_PORT |= LCD_SDI_PIN;
            }
            else
            {
                LCD_SDI_PORT &= ~LCD_SDI_PIN;
            }
            LCD_CLK_PORT |= LCD_CLK_PIN;
            temp <<= 1;
            LCD_CLK_PORT &= ~LCD_CLK_PIN;
        }
        //ä¸‰ä¸ªå­—èŠ‚ä¹‹é—´ä¸€å®šè¦æœ‰è¶³å¤Ÿçš„å»¶æ—¶ï¼Œå¦åˆ™æ˜“å‡ºç°æ—¶åºé—®é¢˜
        delay_Nus( 72 );
    }
    LCD_SDI_PORT &= ~LCD_SDI_PIN;
    LCD_CS_PORT &= ~LCD_CS_PIN;
}
